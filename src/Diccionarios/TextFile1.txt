    Private Function castGriego(ByVal unaLetra As Char) As Char
        Dim caracter As Char
        Select Case unaLetra
            Case "a", "A"
                caracter = ChrW("&H03C9")
            Case "b"
                caracter = ChrW("&h03ca")

        End Select
        Return caracter
    End Function

        If e.KeyChar <> "" Then
            e.Handled = True
            TextBox1.Text = TextBox1.Text & castGriego(e.KeyChar)
        End If




    End Sub
